### Tutorial

Splay Trees are a type of self-balancing binary search tree that has the unique property that more commonly accessed nodes are more quickly retrievable. Like other BSTs, they achieve an amortized $\mathcal{O}(\log n)$ complexity for inserting, deleting, and searching. 

#### Structure

Because Splay Trees are BSTs, they satisfy the **binary search tree property** that for a given node $u$, all nodes $v$ in $u$'s left subtree satisfy $\texttt{key}(v) < \texttt{key}(u)$. Similarly, all nodes $w$ in $u$'s right subtree satisfy $\texttt{key}(u)$ < $\texttt{key}(w)$. 

One form of balancing a splay tree is a **rotation**, where a child of some node $u$ becomes the root of that splay tree. When the left child becomes the new root, the rotation is a right rotation, or a **`zig`** operation. When the right child becomes the new root, the rotation is a left rotation, or a **`zag`** operation. The key idea of a splay tree is to bring some node $u$ to the top of the tree through rotations, while simultaneously balancing the tree.

<!-- Insert Image of Zig/Zag -->

Define an operation called a **`splay-step`** on some node $u$. Let $t$ be the parent of $u$ (if $u$ is not the root), and let $s$ be the parent of $t$ (if $t$ exists and is not the root). The goal of the splay step is to move $u$ to the position where $t$ is (that is, up two levels), without unbalancing the tree. The splay-step can be disected into the following cases:
- **Case 0:** If $u$ is already the root, there is nothing to be done
- **Case 1:** If $t$ is the root, then perform a single `zig` or `zag` operation, depending on which subtree of $t$ $u$ is in.
- **Case 2:** Consider the case where both $t$ and $u$ are both left-children. First perform a `zig` on $s$ and $t$, then perform another `zig` on $t$ and $u$. Similarly, perform two `zag` operations if $t$ and $u$ are right children. 

<!-- Insert Image -->

- **Case 3:** Consider the case where $t$ is a right child and $u$ is a left child. First perform a `zag` on $t$ and $u$, then perform a `zig` on $s$ and $u$. Likewise, do `zig` and then `zag` if $t$ is a left child and $u$ is a right child. 

<!-- Insert Image -->

Finally, we define a **`splay`** operation to be the repeated iteration of `splay-step` until $u$ is at the root. In pseudocode, it looks like:
```
function splay(u):
	while u is not the root:
		splay-step(u)
```

#### Operations
- **find** - _Find an element $x$ in a given splay tree_

  Find the element by walking down the splay tree (like finding an element in a regular BST). When we reach the element, run `splay` on that node. If the element is not present, we still `splay` the last visited node.
- **merge** - _Given two trees $T_1$ and $T_2$ such that $\texttt{key}(u) < \texttt{key}(v) \; \forall u \in T_1, v \in T_2$, create a tree $T$ containing every element of both trees._

  Let $T = T_1$. Find the maximum element of $T$ be repeatedly going to the right child of the current node, starting at the root. Append $T_2$ to the child of the maximum node. 
- **split** - _Given a splay tree $T$ and a pivot $x$, the split operation partitions $T$ into two trees such that $T_1 = \{u \; | \;u \in T \text{ and } \texttt{key}(u) \leq x \}$, $T_2 = \{u \; | \;u \in T \text{ and } \texttt{key}(u) > x \}$_

  Call `find` on $x$. The left subtree of the new root will be $T_2$. Disconnect the edge to its parent. $T$ can be assigned to $T_1$. 
- **insert** - _Given an element $x$, insert it into a given splay tree $T$_

  Create a new splay tree $X$ of one element $x$. Split $T$ with regard to $x$ into $T_1$ and $T_2$, and perform two merges, first with $T_1$ and $X$ and then the result of that `merge` with $T_2$.


- **delete** - _Given an element $x$, delete it from a given splay tree $T$._

  Call `find` on $x$ in $T$. If $x$ exists with $T$, it will be located at the root. We can disconnect its two children and `merge` them together, thus deleteing $x$ from the tree. 

All of the operations take $\mathcal{O}(\log n)$ time.

```cpp
#include <bits/stdc++.h>
using namespace std;

//BeginCodeSnip{Link Cut Tree}
struct Node {
	int x;
	Node *l = 0;
	Node *r = 0;
	Node *p = 0;
	bool rev = false;

	Node() = default;

	Node(int v) {
		x = v;
	}

	void push() {
		if(rev) {
			rev = false;
			swap(l, r);
			if(l) l->rev ^= true;
			if(r) r->rev ^= true;
		}
	}

	bool is_root() {
		return p == 0 || (p->l != this && this != p->r);
	}
};

struct LCT {
	vector<Node> a;

	LCT(int n) {
		a.resize(n+1);
		for(int i = 1; i <= n; ++i)
		a[i].x = i;
	}

	void rot(Node* c) {
		auto p = c->p;
		auto g = p->p;

		if(!p->is_root())
			(g->r == p ? g->r : g->l) = c;

		p->push();
		c->push();

		if(p->l == c) { // rtr
			p->l = c->r;
			c->r = p;
			if(p->l) p->l->p = p;
		} else { // rtl
			p->r = c->l;
			c->l = p;
			if(p->r) p->r->p = p;
		}

		p->p = c;
		c->p = g;
	}

	void splay(Node* c) {
		while(!c->is_root()) {
			auto p = c->p;
			auto g = p->p;
			if(!p->is_root())
				rot((g->r == p) == (p->r == c) ? p : c);
			rot(c);
		}
		c->push();
	}

	Node* access(int v) {
		Node* last = 0;
		Node* c = &a[v];
		for(Node* p = c; p; p = p->p) {
			splay(p);
			p->r = last;
			last = p;
		}
		splay(c);
		return last;
	}

	void make_root(int v) {
		access(v);
		auto* c = &a[v];
		if(c->l)
			c->l->rev ^= true, c->l = 0;
	}

	void link(int u, int v) {
		make_root(v);
		Node* c = &a[v];
		c->p = &a[u];
	}

	void cut(int u, int v) {
		make_root(u);
		access(v);
		if(a[v].l) {
			a[v].l->p = 0;
			a[v].l = 0;
		}
	}

	bool connected(int u, int v) {
		access(u);
		access(v);
		return a[u].p;
	}
};
//EndCodeSnip

int N, M;

int main() {
	cin.tie(0)->sync_with_stdio(0);

	cin >> N >> M;

	LCT lc(N);

	while(M--) {
		string a; cin >> a;
		int b, c; cin >> b >> c;
		if(a == "add") {
			lc.link(b, c);
		}

		if(a == "rem") {
			lc.cut(b, c);
		}

		if(a == "conn") {
			cout << (lc.connected(b, c) ? "YES" : "NO") << "\n";
		}
	}
}
```

### With Euler-Tour Tree

```cpp
// CodeSnip{Benq Template}
//BeginCodeSnip{Euler Tour Tree}
/**
 * Description: Euler Tour Tree using treap, each edge is
 	* represented by two nodes. Supports reroot, insert edge,
 	* delete edge, get connected component. Can support
 	* aggregate over vertices in connected component by
 	* introducing self-loops.
 * Time: O(\log N)
 * Source:
 	* https://codeforces.com/blog/entry/53265 (Rerooting dynamic Euler tour trees)
 	* https://codeforces.com/blog/entry/18369 (On Euler tour trees)
 * Verification:
 	* https://www.spoj.com/problems/DYNACON1/
 */

// TREAP OPERATIONS
int cnt, pri[MX], par[MX]; // ETT node for each edge
AR<int,2> c[MX];

int getRoot(int x) { // get top node in ETT
	while (par[x]) x = par[x];
	return x;
}
void link(int x, int d, int y) { // set d-th child of x to y
	assert(x); assert(d == 0 || d == 1);
	assert(!c[x][d]), c[x][d] = y;
	if (y) assert(!par[y]), par[y] = x;
}
int dis(int x, int d) { // disconnected d-th child of x
	assert(x); assert(d == 0 || d == 1);
	int y = c[x][d]; c[x][d] = 0;
	if (y) assert(par[y] == x), par[y] = 0;
	return y;
}
pi split(int x) { // x and everything to right goes in p.s
	// everything else goes in p.f
	pi p = {dis(x,0),x};
	while (par[x]) {
		int y = par[x];
		if (c[y][0] == x) {
			dis(y,0), link(y,0,p.s),  p.s = y;
		} else {
			assert(c[y][1] == x);
			dis(y,1), link(y,1,p.f); p.f = y;
		}
		x = y;
	}
	assert(!par[p.f] && !par[p.s]); return p;
}

int merge(int x, int y) {
	assert(!par[x] && !par[y]);
	if (!x || !y) return max(x,y);
	if (pri[x] > pri[y]) {
		int X = dis(x,1);
		link(x,1,merge(X,y));
		return x;
	} else {
		int Y = dis(y,0);
		link(y,0,merge(x,Y));
		return y;
	}
}

// int getFirst(int x) {
// 	if (!x) return 0;
// 	while (c[x][0]) x = c[x][0];
// 	return x;
// }
int makeFirst(int x) { // rotate ETT of x such that x is first
	assert(x); pi p = split(x);
	return merge(p.s,p.f);
}
void remFirst(int x) { // remove first node of ETT rooted at x
	assert(x && !par[x]);
	while (c[x][0]) x = c[x][0];
	int y = dis(x,1), p = par[x];
	if (p) dis(p,0), link(p,0,y);
}

// ETT OPERATIONS
map<int,int> adj[MX];
int makeEdge(int a, int b) {
	adj[a][b] = ++cnt; pri[cnt] = rng();
	return cnt;
}
int reroot(int x) { // make edge beginning with x
	if (!sz(adj[x])) return 0;
	return makeFirst(begin(adj[x])->s);
}
bool con(int a, int b) {
	if (!sz(adj[a]) || !sz(adj[b])) return 0;
	a = begin(adj[a])->s, b = begin(adj[b])->s;
	return getRoot(a) == getRoot(b);
}
void add(int a, int b) { // connect A and B with edge
	int ta = reroot(a), tb = reroot(b); if (ta||tb) assert(ta != tb);
	int x = makeEdge(a,b), y = makeEdge(b,a); // make two nodes for new edge
	merge(merge(ta,x),merge(tb,y));
}
void rem(int a, int b) {
	int x = adj[a][b], y = adj[b][a]; makeFirst(x);
	pi p = split(y); // assert(getFirst(p.f) == x && getFirst(p.s) == y);
	remFirst(p.f), remFirst(p.s);
	adj[a].erase(b), adj[b].erase(a);
}
//EndCodeSnip

int main() {
	int N,M; re(N,M);
	F0R(i,M) {
		str s; int A,B; re(s,A,B);
		if (s == "add") {
			add(A,B);
		} else if (s == "rem") {
			rem(A,B);
		} else {
			if (con(A,B)) ps("YES");
			else ps("NO");
		}
	}
}
```
